\section{Key Findings}

\subsection{32-View Results (Table 1)}

\subsubsection{Overall Performance}
\begin{itemize}
    \item DA3 significantly outperforms VGGT across all datasets:
    \begin{itemize}
        \item ETH3D: 85.7\% improvement in AUC3, 27.7\% in F1
        \item ScanNet++: 42.0\% improvement in AUC3, 21.4\% in F1
        \item 7-Scenes: 22.1\% improvement in AUC3, 9.6\% in F1
        \item HiRoom: 63.6\% improvement in AUC3, 53.2\% in F1
    \end{itemize}
\end{itemize}

\subsubsection{FreeGeo Adaptation Results}
\begin{itemize}
    \item \textbf{VGGT+FreeGeo}: 0.54\% weighted improvement over VGGT baseline
    \begin{itemize}
        \item ETH3D: +1.76\% AUC3, -0.14\% F1
        \item ScanNet++: +1.24\% AUC3, -0.07\% F1
        \item 7-Scenes: -0.49\% AUC3, -1.16\% F1 (slight degradation)
        \item HiRoom: +7.09\% AUC3, +6.89\% F1 (best improvement)
    \end{itemize}

    \item \textbf{DA3+FreeGeo}: 2.54\% weighted improvement over DA3 baseline
    \begin{itemize}
        \item ETH3D: +8.38\% AUC3, +0.77\% F1 (largest gain)
        \item ScanNet++: -0.08\% AUC3, -0.00\% F1 (minimal change)
        \item 7-Scenes: +1.06\% AUC3, +1.42\% F1
        \item HiRoom: +3.29\% AUC3, +1.15\% F1
    \end{itemize}
\end{itemize}

\subsubsection{Analysis}
\begin{itemize}
    \item DA3+FreeGeo shows 4.7× larger improvement than VGGT+FreeGeo (2.54\% vs 0.54\%)
    \item FreeGeo adaptation is most effective on ETH3D for both models
    \item HiRoom shows strong gains for VGGT+FreeGeo but moderate gains for DA3+FreeGeo
    \item 7-Scenes shows slight degradation for VGGT+FreeGeo but improvement for DA3+FreeGeo
    \item DA3 baseline already achieves strong performance, leaving less room for FreeGeo improvement
\end{itemize}

\section{Technical Notes}

\subsection{Experimental Setup}
\begin{itemize}
    \item All results are averaged over 3 random seeds (43, 44, 45)
    \item Standard deviation computed with Bessel's correction (ddof=1)
    \item Weighted averages use scene counts: ETH3D=11, ScanNet++=5, 7-Scenes=7, HiRoom=1
    \item Total of 24 scenes across all datasets
\end{itemize}

\subsection{Metric Definitions}
\begin{itemize}
    \item \textbf{AUC3}: Area Under Curve at 3cm threshold for pose estimation accuracy
    \item \textbf{AUC30}: Area Under Curve at 30cm threshold for pose estimation accuracy
    \item \textbf{F1}: Harmonic mean of precision and recall for reconstruction quality
    \item \textbf{CD (Chamfer Distance)}: Overall reconstruction error (lower is better)
\end{itemize}

\subsection{Data Sources}
\begin{itemize}

    \item VGGT+FreeGeo: \texttt{workspace/vggt\_final\_benchmark/}

    \item DA3+FreeGeo: \texttt{workspace/da3\_final\_benchmark/}
\end{itemize}

\subsection{Weighted Average Calculation}
The weighted average improvement is calculated as:
\begin{equation}
\text{Weighted Avg} = \frac{1}{2}\left(\sum_{d \in D} w_d \cdot \frac{\Delta \text{AUC3}_d}{\text{AUC3}_{d,\text{baseline}}} + \sum_{d \in D} w_d \cdot \frac{\Delta \text{F1}_d}{\text{F1}_{d,\text{baseline}}}\right) \times 100\%
\end{equation}
where $w_d = \frac{n_d}{\sum_{d'} n_{d'}}$ is the weight based on scene count $n_d$ for dataset $d$.
